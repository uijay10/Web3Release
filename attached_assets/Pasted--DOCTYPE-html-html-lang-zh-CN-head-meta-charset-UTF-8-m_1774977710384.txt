    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>AI 项目事件聚合</title>
    <script src="https://cdn.tailwindcss.com"></script>
    <script>
        tailwind.config = { content: [], theme: { extend: {} } };
    </script>
</head>
<body class="bg-slate-50 font-sans">
    <header class="bg-gradient-to-r from-blue-600 to-indigo-700 text-white py-10 shadow-lg">
        <div class="max-w-5xl mx-auto px-6 text-center">
            <h1 class="text-4xl font-bold">AI 项目事件聚合</h1>
            <p class="mt-3 text-blue-100 text-lg">自动抓取 · 轻量认领 · 需求匹配</p>
        </div>
    </header>

    <nav class="bg-white shadow sticky top-0 z-50 border-b">
        <div class="max-w-5xl mx-auto px-6 py-5 flex flex-wrap gap-3 justify-center" id="nav-bar"></div>
    </nav>

    <div class="max-w-4xl mx-auto px-6 py-8">
        <div class="flex justify-between items-center mb-6">
            <span id="result-count" class="text-slate-500 text-sm">共 0 条事件</span>
        </div>
        <div id="event-list" class="space-y-6"></div>
    </div>

    <script>
        let allEvents = [];

        const categories = [
            "全部", "测试网", "IDO/Launchpad", "预售", "融资公告", 
            "空投", "招聘", "节点招募", "主网上线", 
            "代币解锁", "交易所上线", "链上任务", "开发者专区"
        ];

        function createNavBar() {
            const nav = document.getElementById('nav-bar');
            nav.innerHTML = '';

            categories.forEach((cat, index) => {
                const btn = document.createElement('button');
                btn.textContent = cat;
                btn.dataset.category = cat;
                btn.className = `px-6 py-2.5 rounded-full text-sm font-medium transition-all ${
                    index === 0 
                        ? 'bg-blue-600 text-white shadow-md' 
                        : 'bg-white hover:bg-slate-100 border border-slate-200'
                }`;

                btn.addEventListener('click', () => {
                    // 更新按钮样式
                    document.querySelectorAll('#nav-bar button').forEach(b => {
                        b.classList.remove('bg-blue-600', 'text-white', 'shadow-md');
                        b.classList.add('bg-white', 'border', 'border-slate-200');
                    });
                    btn.classList.add('bg-blue-600', 'text-white', 'shadow-md');
                    btn.classList.remove('bg-white', 'border', 'border-slate-200');

                    filterEvents(cat);
                });

                nav.appendChild(btn);
            });
        }

        function filterEvents(selectedCategory) {
            const container = document.getElementById('event-list');
            container.innerHTML = '';

            let filtered = allEvents;

            if (selectedCategory !== '全部') {
                filtered = allEvents.filter(event => {
                    const cats = Array.isArray(event.category) ? event.category : [event.category || ''];
                    return cats.some(c => c === selectedCategory);
                });
            }

            document.getElementById('result-count').textContent = `共 ${filtered.length} 条事件`;

            if (filtered.length === 0) {
                container.innerHTML = `<div class="text-center py-16 text-slate-500">暂无「${selectedCategory}」相关事件</div>`;
                return;
            }

            renderEvents(filtered);
        }

        function renderEvents(events) {
            const container = document.getElementById('event-list');
            container.innerHTML = '';

            events.forEach(event => {
                const cats = Array.isArray(event.category) ? event.category : [event.category || '其他'];
                const mainCat = cats[0] || '其他';

                const desc = event.description || event.content || '暂无描述';
                const shortDesc = desc.length > 220 ? desc.substring(0, 217) + '...' : desc;

                const cardHTML = `
                    <div class="bg-white rounded-3xl shadow-sm hover:shadow-xl transition-all p-7 border border-slate-100">
                        <div class="flex justify-between items-start mb-5">
                            <div class="flex gap-2 flex-wrap">
                                <span class="px-5 py-1.5 bg-blue-100 text-blue-700 text-xs font-semibold rounded-2xl">${mainCat}</span>
                                ${event.importance ? `<span class="px-4 py-1 text-xs font-medium rounded-2xl ${event.importance.includes('重要') ? 'bg-red-100 text-red-600' : 'bg-amber-100 text-amber-600'}">${event.importance}</span>` : ''}
                            </div>
                            <span class="text-slate-500 text-sm whitespace-nowrap">${event.start_time || '日期待定'}</span>
                        </div>

                        <h2 class="text-2xl font-semibold leading-tight mb-4">
                            <a href="${event.source_url || '#'}" target="_blank" class="hover:text-blue-600">${event.title}</a>
                        </h2>

                        <p class="text-slate-600 leading-relaxed mb-6">${shortDesc}</p>

                        <div class="flex flex-wrap gap-2 mb-6">
                            ${Array.isArray(event.tags) ? event.tags.map(tag => 
                                `<span class="px-3 py-1 bg-slate-100 text-slate-600 text-xs rounded-xl">#${tag}</span>`
                            ).join('') : ''}
                        </div>

                        <div class="flex justify-between items-center pt-4 border-t text-sm">
                            <span class="text-slate-500">来源：${event.source || 'Solana'}</span>
                            <a href="${event.source_url || '#'}" target="_blank" class="text-blue-600 hover:underline font-medium">查看详情 →</a>
                        </div>
                    </div>
                `;
                container.innerHTML += cardHTML;
            });
        }

        async function loadEvents() {
            try {
                const res = await fetch('/extraction_result.json');
                if (!res.ok) throw new Error('加载失败');
                
                allEvents = await res.json();
                console.log('加载成功，共', allEvents.length, '条');

                // 默认显示全部
                document.getElementById('result-count').textContent = `共 ${allEvents.length} 条事件`;
                renderEvents(allEvents);
            } catch (err) {
                console.error(err);
                document.getElementById('event-list').innerHTML = `<div class="text-red-500 text-center py-20">数据加载失败，请检查 JSON 文件</div>`;
            }
        }

        window.onload = () => {
            createNavBar();
            loadEvents();
        };
    </script>
</body>
