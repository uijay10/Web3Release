                        class="px-6 py-2.5 bg-blue-600 hover:bg-blue-700 text-white rounded-2xl transition-colors">
                    刷新数据

        <!-- 统计卡片 -->
        <div class="max-w-7xl mx-auto px-8 py-8 grid grid-cols-1 md:grid-cols-3 gap-6">

                <div class="text-amber-600 text-sm font-medium">待审核</div>
                <div id="pending-count" class="text-5xl font-bold text-slate-800 mt-3">0</div>
            </div>
            <div class="bg-white rounded-3xl p-6 shadow-sm border border-slate-100">

            <div class="bg-white rounded-3xl p-6 shadow-sm border border-slate-100">
                <div class="text-red-600 text-sm font-medium">已拒绝</div>

                <input

                    type="text" 
                    placeholder="搜索项目名称或事件标题..." 
                    class="w-96 px-5 py-3 border border-slate-300 rounded-3xl focus:outline-none focus:border-blue-500 text-base">
            </div>

    <script>
        let claims = [
            {
                id: 1,
                eventTitle: "Colosseum Frontier 黑客松",
                projectName: "Solana 中文社区",
                contact: "@solana_cn",
                reason: "希望作为社区方提供中文直播、开发者招募支持以及线下 Meetup 组织。",
                eventLink: "#",
                status: "pending",
                submitTime: "2026-04-01 14:30",
                reviewTime: ""
            },
            {
                id: 2,
                eventTitle: "Solana Accelerate USA",
                projectName: "Ocean Ventures",
                contact: "hello@ocean.ventures",
                reason: "希望作为赞助方参与线下活动，并提供投资人资源对接。",
                eventLink: "#",
                status: "approved",
                submitTime: "2026-03-28 09:15",
                reviewTime: "2026-03-29 10:20"
            }
        ];

        function getStatusHTML(status) {
            if (status === 'pending') {
                return `<span class="status-pending px-5 py-1.5 rounded-2xl text-sm font-medium">待审核</span>`;
            } else if (status === 'approved') {
                return `<span class="status-approved px-5 py-1.5 rounded-2xl text-sm font-medium">已通过</span>`;
            } else {
                return `<span class="status-rejected px-5 py-1.5 rounded-2xl text-sm font-medium">已拒绝</span>`;
            }
        }

        function renderClaims(filteredClaims) {
            const container = document.getElementById('claims-list');
            container.innerHTML = '';

            document.getElementById('total-count').textContent = claims.length;

            const pending = claims.filter(c => c.status === 'pending').length;
            const approved = claims.filter(c => c.status === 'approved').length;
            const rejected = claims.filter(c => c.status === 'rejected').length;

            document.getElementById('pending-count').textContent = pending;
            document.getElementById('approved-count').textContent = approved;
            document.getElementById('rejected-count').textContent = rejected;

            filteredClaims.forEach(claim => {
                const html = `
                    <div class="claim-card bg-white rounded-3xl p-8 shadow-sm border border-slate-100">
                        <div class="flex justify-between items-start">
                            <div class="flex-1 min-w-0">
                                <div class="flex items-center gap-3">
                                    <h3 class="font-semibold text-xl text-slate-900">${claim.eventTitle}</h3>
                                    ${getStatusHTML(claim.status)}
                                </div>
                                <p class="mt-3 text-slate-700"><strong>申请方：</strong>${claim.projectName}</p>
                                <p class="text-slate-600"><strong>联系方式：</strong>${claim.contact}</p>
                                <p class="text-xs text-slate-500 mt-1">提交时间：${claim.submitTime}</p>
                                ${claim.reviewTime ? `<p class="text-xs text-slate-500">审核时间：${claim.reviewTime}</p>` : ''}
                            </div>
                        </div>

                        <div class="mt-6 bg-slate-50 p-6 rounded-2xl text-slate-700">
                            <strong class="block mb-2">认领理由：</strong>
                            ${claim.reason}
                        </div>

                        <div class="mt-8 flex gap-4">
                            <a href="${claim.eventLink}" target="_blank" 
                               class="text-blue-600 hover:text-blue-700 font-medium">查看原事件 →</a>
                            
                            ${claim.status === 'pending' ? `
                                <button onclick="approveClaim(${claim.id})" 
                                        class="px-8 py-3 bg-emerald-600 hover:bg-emerald-700 text-white rounded-2xl transition-colors">通过审核</button>
                                <button onclick="rejectClaim(${claim.id})" 
                                        class="px-8 py-3 bg-red-600 hover:bg-red-700 text-white rounded-2xl transition-colors">拒绝申请</button>
                            ` : ''}
                        </div>
                    </div>
                `;
                container.innerHTML += html;
            });
        }

        function approveClaim(id) {
            const claim = claims.find(c => c.id === id);
            if (claim) {
                claim.status = 'approved';
                claim.reviewTime = new Date().toLocaleString('zh-CN');
                alert(`✅ 已通过 ${claim.projectName} 的认领申请`);
                renderClaims(claims);
            }
        }

        function rejectClaim(id) {
            const claim = claims.find(c => c.id === id);
            if (claim) {
                claim.status = 'rejected';
                claim.reviewTime = new Date().toLocaleString('zh-CN');
                alert(`❌ 已拒绝 ${claim.projectName} 的认领申请`);
                renderClaims(claims);
            }
        }

        function refreshData() {
            alert("实际项目中此处会从 Supabase 等后端拉取最新数据");
            renderClaims(claims);
        }

        // 搜索功能
        document.getElementById('search-input').addEventListener('input', (e) => {
            const term = e.target.value.toLowerCase().trim();
            const filtered = claims.filter(c => 
                c.projectName.toLowerCase().includes(term) || 
                c.eventTitle.toLowerCase().includes(term)
            );
            renderClaims(filtered);
        });

        // 初始化
        window.onload = () => {
            renderClaims(claims);
        };
    </script>
