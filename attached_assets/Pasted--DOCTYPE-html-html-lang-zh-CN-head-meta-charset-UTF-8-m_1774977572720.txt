        <p class="mt-2">自动抓取 · 轻量认领 · 需求匹配</p>

    <div class="max-w-4xl mx-auto px-6 py-10">
        <div id="status" class="text-center py-10 text-lg">正在加载数据...</div>
        <div id="event-list" class="space-y-6"></div>
    </div>

    <script>
        async function loadEvents() {
            const status = document.getElementById('status');
            try {
                const res = await fetch('/extraction_result.json');
                if (!res.ok) throw new Error('文件未找到');
                
                const events = await res.json();
                status.textContent = `成功加载 ${events.length} 条事件`;
                
                const container = document.getElementById('event-list');
                container.innerHTML = '';

                events.forEach(event => {
                    const card = document.createElement('div');
                    card.className = "bg-white p-6 rounded-2xl shadow mb-6";
                    card.innerHTML = `
                        <h2 class="text-xl font-bold mb-2">${event.title || '无标题'}</h2>
                        <p class="text-gray-600">${(event.description || '').substring(0, 150)}...</p>
                        <p class="text-sm text-blue-600 mt-3">日期: ${event.start_time || '未知'}</p>
                    `;
                    container.appendChild(card);
                });
            } catch (e) {
                status.innerHTML = `<span class="text-red-600">加载失败: ${e.message}</span>`;
                console.error(e);
            }
        }

        window.onload = loadEvents;
    </script>
