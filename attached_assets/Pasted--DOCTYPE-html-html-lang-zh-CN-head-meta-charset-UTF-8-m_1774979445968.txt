            <div class="space-y-5">
                <div>
                    <label class="block text-sm font-medium mb-1">项目/团队名称 <span class="text-red-500">*</span></label>
                    <input id="project-name" type="text" class="w-full px-4 py-3 border border-slate-300 rounded-2xl focus:outline-none focus:border-purple-500" placeholder="您的项目名称">
                </div>
                <div>
                    <label class="block text-sm font-medium mb-1">联系方式 <span class="text-red-500">*</span></label>
                    <input id="contact-info" type="text" class="w-full px-4 py-3 border border-slate-300 rounded-2xl focus:outline-none focus:border-purple-500" placeholder="Telegram / Email / Twitter">
                </div>
                <div>
                    <label class="block text-sm font-medium mb-1">认领理由 / 合作意向</label>
                    <textarea id="claim-reason" rows="4" class="w-full px-4 py-3 border border-slate-300 rounded-3xl focus:outline-none focus:border-purple-500" placeholder="请简要说明认领理由..."></textarea>
                </div>

            <div class="flex gap-3 mt-8">
                <button onclick="closeModal()" class="flex-1 py-3 border border-slate-300 rounded-2xl hover:bg-slate-50">取消</button>
                <button id="submit-btn" onclick="submitProjectClaim()" 
                        class="flex-1 py-3 bg-purple-600 text-white rounded-2xl hover:bg-purple-700">提交认领申请</button>
            </div>
        </div>

    <!-- Toast 提示 -->
    <div id="toast" class="hidden fixed bottom-6 right-6 bg-slate-800 text-white px-6 py-4 rounded-2xl shadow-2xl flex items-center gap-3 z-[200]">
        <span id="toast-icon" class="text-xl"></span>
        <span id="toast-message" class="text-sm"></span>
    </div>

    <script>
        // ... 你的其他代码（导航、搜索、渲染卡片等）保持不变 ...

        let currentEventTitle = '';

        function openProjectClaimModal(title) {
            currentEventTitle = title;
            document.getElementById('modal-event-title').value = title;
            document.getElementById('claim-modal').classList.remove('hidden');
            document.getElementById('project-name').value = '';
            document.getElementById('contact-info').value = '';
            document.getElementById('claim-reason').value = '';
        }

        function closeModal() {
            document.getElementById('claim-modal').classList.add('hidden');
        }

        function showToast(message, success = true) {
            const toast = document.getElementById('toast');
            document.getElementById('toast-icon').textContent = success ? '✅' : '⚠️';
            document.getElementById('toast-message').textContent = message;
            toast.classList.remove('hidden');
            setTimeout(() => toast.classList.add('hidden'), 4500);
        }

        function submitProjectClaim() {
            const btn = document.getElementById('submit-btn');
            const projectName = document.getElementById('project-name').value.trim();
            const contact = document.getElementById('contact-info').value.trim();
            const reason = document.getElementById('claim-reason').value.trim();

            if (!projectName || !contact) {
                showToast("请填写项目名称和联系方式", false);
                return;
            }

            btn.disabled = true;
            btn.textContent = "提交中...";

            // 保存到 localStorage（供后台读取）
            let applications = JSON.parse(localStorage.getItem('projectApplications') || '[]');
            applications.unshift({
                id: Date.now(),
                eventTitle: currentEventTitle,
                projectName: projectName,
                contact: contact,
                reason: reason || "未填写",
                status: "pending",
                submitTime: new Date().toLocaleString('zh-CN'),
                reviewNote: ""
            });
            localStorage.setItem('projectApplications', JSON.stringify(applications));

            setTimeout(() => {
                showToast("✅ 申请已提交，等待管理员审核");
                closeModal();
                btn.disabled = false;
                btn.textContent = "提交认领申请";
            }, 800);
        }

        // 打开后台时打印当前申请数量（调试用）
        console.log("当前待审核项目方申请数量：", JSON.parse(localStorage.getItem('projectApplications') || '[]').length);
    </script>
